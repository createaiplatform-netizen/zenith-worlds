import streamlit as st import alpaca_trade_api as tradeapi st.title("📊 Trading Dashboard") # ========================= # SAFE SECRET LOADING # ========================= try: API_KEY = st.secrets["APCA_API_KEY_ID"] API_SECRET = st.secrets["APCA_API_SECRET_KEY"] BASE_URL = st.secrets.get("BASE_URL", "https://paper-api.alpaca.markets") api = tradeapi.REST( API_KEY, API_SECRET, base_url=BASE_URL, api_version="v2" ) st.success("✅ Alpaca connected") except Exception as e: st.error("❌ Connection issue") st.write(e) api = None # ========================= # UI ALWAYS RENDERS # ========================= st.subheader("System Status") if api: try: account = api.get_account() st.write("Account Status:", account.status) st.write("Buying Power:", account.buying_power) st.write("Cash:", account.cash) except Exception as e: st.error("Failed to fetch account data") st.write(e) else: st.warning("API not initialized") st.button("Refresh")
